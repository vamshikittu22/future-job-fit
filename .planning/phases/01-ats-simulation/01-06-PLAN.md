```typescript
function convertResumeToText(resume: Resume): string {
  // Convert structured resume data to plain text
  // similar to what ATS would extract
  const sections: string[] = [];
  
  // Personal info
  if (resume.personal) {
    sections.push(`${resume.personal.firstName} ${resume.personal.lastName}`);
    sections.push(resume.personal.email || '');
    sections.push(resume.personal.phone || '');
  }
  
  // Summary
  if (resume.summary) {
    sections.push('SUMMARY');
    sections.push(resume.summary);
  }
  
  // Experience
  if (resume.experience?.length) {
    sections.push('EXPERIENCE');
    resume.experience.forEach(exp => {
      sections.push(`${exp.title} at ${exp.company}`);
      sections.push(`${exp.startDate} - ${exp.endDate || 'Present'}`);
      sections.push(exp.description || '');
    });
  }
  
  // Education
  if (resume.education?.length) {
    sections.push('EDUCATION');
    resume.education.forEach(edu => {
      sections.push(`${edu.degree} from ${edu.institution}`);
      sections.push(`${edu.startDate} - ${edu.endDate || 'Present'}`);
    });
  }
  
  // Skills
  if (resume.skills?.length) {
    sections.push('SKILLS');
    resume.skills.forEach(category => {
      sections.push(`${category.name}: ${category.items.join(', ')}`);
    });
  }
  
  return sections.filter(Boolean).join('\n\n');
}
```
