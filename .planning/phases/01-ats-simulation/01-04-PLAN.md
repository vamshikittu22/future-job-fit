```typescript
export function simulateWorkday(resumeText: string, layout?: LayoutAnalysis): ParsedResumeResult {
  // 1. Extract sections with strict header matching
  const sections = extractSections(resumeText, {
    strictHeaders: WORKDAY_QUIRKS.strictHeaders,
    allowedHeaders: WORKDAY_QUIRKS.allowedHeaders
  });
  
  // 2. Apply table penalty if detected
  let qualityPenalty = 1.0;
  if (layout?.tables.some(t => t.severity === 'critical' || t.severity === 'high')) {
    qualityPenalty *= (1 - WORKDAY_QUIRKS.tablePenalty);
  }
  
  // 3. Apply column penalty
  if (layout?.columns.some(c => c.type === 'float')) {
    qualityPenalty *= (1 - WORKDAY_QUIRKS.columnPenalty);
  }
  
  // 4. Check date format compliance
  const dateIssues: DataLossWarning[] = [];
  sections.forEach(section => {
    section.fields.forEach(field => {
      if (field.fieldType === 'date_range' && !isStandardDateFormat(field.value)) {
        dateIssues.push({
          type: 'date_unparsed',
          severity: 'medium',
          originalText: field.value,
          extractedText: null,
          section: section.title
        });
      }
    });
  });
  
  // 5. Calculate extraction quality
  const baseQuality = calculateAverageConfidence(sections);
  const extractionQuality = Math.round(baseQuality * qualityPenalty);
  
  return {
    platform: 'WORKDAY',
    sections,
    extractionQuality,
    dataLossWarnings: dateIssues,
    parsingErrors: []
  };
}
```

```typescript
export const TALEO_QUIRKS: PlatformQuirks = {
  strictHeaders: true,
  allowedHeaders: [
    'Work Experience', 'Education', 'Skills', 'Summary'
  ],
  tablePenalty: 0.35,
  columnPenalty: 0.5,        // Taleo REALLY struggles with columns
  dateFormatStrict: true,
  supportedDateFormats: ['MM/YYYY', 'YYYY'], // Very limited
  headerDetection: 'strict',
  footerSensitivity: 'high'
};

export function simulateTaleo(resumeText: string, layout?: LayoutAnalysis): ParsedResumeResult {
  const sections = extractSections(resumeText, {
    strictHeaders: true
  });
  
  let qualityPenalty = 1.0;
  
  // Taleo is worst with columns
  if (layout?.columns.some(c => c.type === 'float' || c.type === 'flex')) {
    qualityPenalty *= (1 - TALEO_QUIRKS.columnPenalty);
  }
  
  // And bad with tables
  if (layout?.tables.length) {
    qualityPenalty *= (1 - TALEO_QUIRKS.tablePenalty);
  }
  
  // Limited date format support causes issues
  const dateIssues: DataLossWarning[] = [];
  sections.forEach(section => {
    section.fields
      .filter(f => f.fieldType === 'date_range')
      .forEach(field => {
        if (!/^(\d{1,2}\/\d{2,4}|\d{4})/.test(field.value)) {
          dateIssues.push({
            type: 'date_unparsed',
            severity: 'high',
            originalText: field.value,
            extractedText: null,
            section: section.title
          });
        }
      });
  });
  
  return {
    platform: 'TALEO',
    sections,
    extractionQuality: Math.round(calculateAverageConfidence(sections) * qualityPenalty),
    dataLossWarnings: dateIssues,
    parsingErrors: []
  };
}
```

```typescript
// platforms/index.ts
export { simulateWorkday, WORKDAY_QUIRKS } from './workday';
export { simulateLever, LEVER_QUIRKS } from './lever';
export { simulateGreenhouse, GREENHOUSE_QUIRKS } from './greenhouse';
export { simulateTaleo, TALEO_QUIRKS } from './taleo';

export const PLATFORMS: PlatformType[] = ['WORKDAY', 'LEVER', 'GREENHOUSE', 'TALEO'];

export function simulatePlatform(
  platform: PlatformType, 
  resumeText: string, 
  layout?: LayoutAnalysis
): ParsedResumeResult {
  switch (platform) {
    case 'WORKDAY': return simulateWorkday(resumeText, layout);
    case 'LEVER': return simulateLever(resumeText, layout);
    case 'GREENHOUSE': return simulateGreenhouse(resumeText, layout);
    case 'TALEO': return simulateTaleo(resumeText, layout);
    default: throw new Error(`Unknown platform: ${platform}`);
  }
}

export function comparePlatforms(
  resumeText: string,
  layout?: LayoutAnalysis
): PlatformComparison {
  const results = PLATFORMS.map(platform => ({
    platform,
    result: simulatePlatform(platform, resumeText, layout)
  }));
  
  const scores = results.map(r => ({
    platform: r.platform,
    score: r.result.extractionQuality,
    issues: generateIssues(r.result),
    riskLevel: calculateRiskLevel(r.result.extractionQuality)
  }));
  
  return {
    timestamp: Date.now(),
    results,
    scores,
    bestPlatform: scores.reduce((best, current) => 
      current.score > best.score ? current : best
    ),
    worstPlatform: scores.reduce((worst, current) => 
      current.score < worst.score ? current : worst
    )
  };
}
```
