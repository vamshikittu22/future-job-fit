    3. Calculate weighted score: (Required Match × 0.50) + (Preferred Match × 0.25) + (Experience Match × 0.15) + (Keywords × 0.10)

    ```typescript
    import type { WeightedKeywordResult, WeightedScoreBreakdown, MatchedSkill } from '../types';

    const WEIGHTS = {
      required: 0.50,
      preferred: 0.25,
      bonus: 0.10,
    };

    // Normalize keyword for matching (lowercase, trim)
    const normalize = (s: string) => s.toLowerCase().trim();

    // Check if keyword matches (including partial matches)
    function isKeywordMatched(keyword: string, resumeText: string): boolean {
      const normKeyword = normalize(keyword);
      const normResume = normalize(resumeText);
      
      // Exact match
      if (normResume.includes(normKeyword)) return true;
      
      // Partial match: "python" matches "python script", "js" matches "javascript"
      const variations = [
        { short: 'js', long: 'javascript' },
        { short: 'py', long: 'python' },
        { short: 'ts', long: 'typescript' },
        { short: 'ai', long: 'artificial intelligence' },
        { short: 'ml', long: 'machine learning' },
      ];
      
      for (const v of variations) {
        if ((normKeyword.includes(v.short) && normResume.includes(v.long)) ||
            (normKeyword.includes(v.long) && normResume.includes(v.short))) {
          return true;
        }
      }
      
      return false;
    }

    // Determine match status
    function getMatchStatus(keyword: string, resumeText: string): 'matched' | 'partial' | 'missing' {
      const normKeyword = normalize(keyword);
      const normResume = normalize(resumeText);
      
      if (normResume.includes(normKeyword)) return 'matched';
      if (isKeywordMatched(keyword, resumeText)) return 'partial';
      return 'missing';
    }

    // Calculate weight type based on keyword weight
    export function getWeightType(baseWeight: number): 'required' | 'preferred' | 'bonus' {
      if (baseWeight >= 0.7) return 'required';
      if (baseWeight >= 0.4) return 'preferred';
      return 'bonus';
    }

    // Main function: calculate weighted keywords
    export function calculateWeightedKeywords(
      keywords: Array<{ keyword: string; weight: number; jdSection: string }>,
      resumeText: string
    ): WeightedKeywordResult[] {
      return keywords.map((kw) => {
        const status = getMatchStatus(kw.keyword, resumeText);
        const weightType = getWeightType(kw.weight);
        const baseWeight = WEIGHTS[weightType];
        
        let matchScore: number;
        if (status === 'matched') {
          matchScore = baseWeight * 100;
        } else if (status === 'partial') {
          matchScore = baseWeight * 50;
        } else {
          matchScore = 0;
        }

        return {
          keyword: kw.keyword,
          weightType,
          baseWeight,
          actualWeight: baseWeight * kw.weight,
          status,
          matchScore,
          locations: [], // Would be populated with actual locations in real impl
          sourceJD: kw.jdSection,
        };
      });
    }

    // Calculate breakdown from keyword results
    export function calculateBreakdown(results: WeightedKeywordResult[]): WeightedScoreBreakdown {
      const required = results.filter(r => r.weightType === 'required');
      const preferred = results.filter(r => r.weightType === 'preferred');
      const bonus = results.filter(r => r.weightType === 'bonus');

      const getScore = (items: WeightedKeywordResult[]) => 
        items.reduce((sum, r) => sum + r.matchScore, 0);
      
      const getMax = (items: WeightedKeywordResult[]) =>
        items.reduce((sum, r) => sum + WEIGHTS[r.weightType as keyof typeof WEIGHTS] * 100, 0);

      const overall = getScore(results);
      const maxOverall = getMax(results);

      return {
        requiredSkills: {
          total: required.length,
          matched: required.filter(r => r.status === 'matched').length,
          score: getScore(required),
          maxScore: getMax(required),
        },
        preferredSkills: {
          total: preferred.length,
          matched: preferred.filter(r => r.status === 'matched').length,
          score: getScore(preferred),
          maxScore: getMax(preferred),
        },
        bonusSkills: {
          total: bonus.length,
          matched: bonus.filter(r => r.status === 'matched').length,
          score: getScore(bonus),
          maxScore: getMax(bonus),
        },
        overall: {
          score: overall,
          maxScore: maxOverall,
          percentage: maxOverall > 0 ? (overall / maxOverall) * 100 : 0,
        },
      };
    }

    // Experience match calculation (15% weight)
    export function calculateExperienceScore(
      resumeExperience: string,
      jdExperience: string
    ): number {
      // Simple check: count years mentioned vs required
      const resumeYears = (resumeExperience.match(/\d+\+?\s*years?/gi) || []).length;
      const jdYears = (jdExperience.match(/\d+\+?\s*years?/gi) || []).length;
      
      if (jdYears === 0) return 15; // No experience requirement = full points
      if (resumeYears >= jdYears) return 15;
      if (resumeYears >= jdYears * 0.5) return 10;
      if (resumeYears > 0) return 5;
      return 0;
    }
    ```

    ```typescript
    import { useMemo } from 'react';
    import { useResume } from '@/shared/contexts/ResumeContext';
    import { useJob } from '@/shared/contexts/JobContext';
    import { calculateWeightedKeywords, calculateBreakdown, calculateExperienceScore } from '../utils/keywordWeighting';
    import type { MatchScoreResult, WeightedScoreBreakdown } from '../types';

    interface UseMatchScoreOptions {
      includeExperience?: boolean;
    }

    export function useMatchScore(options: UseMatchScoreOptions = {}): {
      score: number;
      breakdown: WeightedScoreBreakdown | null;
      isLoading: boolean;
      hasJobDescription: boolean;
      keywords: Array<{ keyword: string; weight: number; jdSection: string }>;
    } {
      const { resumeData } = useResume();
      const { currentJob, extractedFields } = useJob();

      const scoreResult = useMemo(() => {
        if (!currentJob?.description) {
          return null;
        }

        // Extract keywords from job description
        // In real implementation, this would use the JD parser
        const keywords = extractedFields?.keywords || [];
        
        // Convert resume data to plain text for matching
        const resumeText = [
          resumeData.personalInfo?.summary,
          ...(resumeData.experience || []).flatMap(exp => [
            exp.title,
            exp.company,
            ...(exp.bullets || [])
          ]),
          ...(resumeData.skills || []).flatMap(s => s.items),
          ...(resumeData.projects || []).flatMap(p => [p.title, p.description]),
        ].filter(Boolean).join(' ');

        // Calculate weighted keywords
        const keywordResults = calculateWeightedKeywords(keywords, resumeText);
        
        // Calculate breakdown
        const breakdown = calculateBreakdown(keywordResults);
        
        // Experience score (15% of total)
        const experienceText = (resumeData.experience || [])
          .map(e => `${e.title} ${e.company} ${e.bullets?.join(' ')}`)
          .join(' ');
        const experienceScore = calculateExperienceScore(experienceText, currentJob.description);

        // Keywords bonus (10% of total)
        const keywordsBonus = breakdown.overall.percentage * 0.10;

        // Calculate overall score
        const weightedComponent = breakdown.overall.percentage * 0.75; // 75% from required/preferred/bonus
        const overallScore = Math.min(100, weightedComponent + experienceScore + keywordsBonus);

        return {
          breakdown,
          keywords: keywordResults,
          overallScore,
          experienceScore,
        };
      }, [resumeData, currentJob, extractedFields]);

      return {
        score: scoreResult?.overallScore ?? 0,
        breakdown: scoreResult?.breakdown ?? null,
        isLoading: false,
        hasJobDescription: Boolean(currentJob?.description),
        keywords: [],
      };
    }

    export default useMatchScore;
    ```
